Private Sub Workbook_Open()
    
    Debug.Print "--- Workbook_Open STARTS ---"
    
    Dim a As Byte
    Dim ArrayOS As String
    Dim RegistryFolder As String
    
    Range("LastOpened") = Format(Now, "ddd dd-mmm-yyyy hh:mm:ss")
    Range("ExcelVersion") = Application.Version
    Range("ExcelBuild") = Application.Build
    
' --- OS section STARTS ---

Range("OS") = Application.OperatingSystem

If Left(Application.OperatingSystem, 3) = "Win" Then

    Debug.Print "Windows detected. Writing additional OS info to the appropriate named ranges."
    
    RegistryFolder = "HKEY_LOCAL_MACHINE\SOFTWARE\Microsoft\Windows NT\CurrentVersion\"
    Debug.Print Replace("RegistryFolder: '" & RegistryFolder & "'.", Chr(39), Chr(34))
    
    ArrayOS = ArrayOS & "MajorOS|CurrentMajorVersionNumber" & vbCr
    ArrayOS = ArrayOS & "MinorOS|CurrentMinorVersionNumber" & vbCr
    ArrayOS = ArrayOS & "BuildOS|CurrentBuild" & vbCr
    ArrayOS = ArrayOS & "UBROS|UBR"
    
    For a = LBound(Split(ArrayOS, vbCr)) To UBound(Split(ArrayOS, vbCr))
        
        LeftOfPipe = Split(Split(ArrayOS, vbCr)(a), "|")(0)
        Debug.Print Replace("LeftOfPipe: '" & LeftOfPipe & "'.", Chr(39), Chr(34))
        
        RightOfPipe = Split(Split(ArrayOS, vbCr)(a), "|")(1)
        Debug.Print Replace("RightOfPipe: '" & RightOfPipe & "'.", Chr(39), Chr(34))
        
        Range(LeftOfPipe) = CreateObject("WScript.Shell").RegRead(RegistryFolder & RightOfPipe)
        
    Next a

    Range("NameMacOS") = "Mac only obvs"

Else
    Debug.Print "macOS detected. Writing macOS version info to the appropriate named ranges."

    ' macOS equivalents using sw_vers
    On Error Resume Next

    Range("MajorOS") = MacScript("do shell script ""sw_vers -productVersion | cut -d. -f1""")
    Range("MinorOS") = MacScript("do shell script ""sw_vers -productVersion | cut -d. -f2""")
    Range("BuildOS") = MacScript("do shell script ""sw_vers -buildVersion""")
    Range("UBROS") = "Windows only" ' MacOS has no UBR equivalent
    Range("NameMacOS") = _
    Replace("=IFERROR(INDEX(NamesMacOSs[NameMacOS],MATCH(MajorOS,NamesMacOSs[MajorOS],0)), 'Error :0(')", Chr(39), Chr(34))

    If Err.Number = 0 Then
    
        Debug.Print "MacScripts worked :0) No need to try AppleScripts instead."
    
    Else ' MacScripts failed
        
        Debug.Print "MacScript failed; Excel sandboxed. Trying AppleScriptTask."

        Err.Clear

        Range("MajorOS") = AppleScriptTask("MacOSVersion.scpt", "GetMajor", "")
        Range("MinorOS") = AppleScriptTask("MacOSVersion.scpt", "GetMinor", "")
        Range("BuildOS") = AppleScriptTask("MacOSVersion.scpt", "GetBuild", "")
        Range("UBROS") = "Windows only"
        Range("NameMacOS") = _
        Replace("=IFERROR(INDEX(NamesMacOSs[NameMacOS],MATCH(MajorOS,NamesMacOSs[MajorOS],0)), 'Error :0(')", Chr(39), Chr(34))
        
    End If

    On Error GoTo 0

End If

    ' --- OS section ENDS ---
    
    Range("Username") = Environ("USERNAME")
    Range("User") = Environ("USER")
    
    ' --- Buttons: Home tab STARTS ---
    
    For b = 1 To Sheets("Home").Buttons.Count
    
        Select Case Left(Application.OperatingSystem, 3)
    
            Case "Mac"
            
                Select Case Sheets("Home").Buttons(b).Caption
                
                    Case "Open containing folder"
                    
                        Sheets("Home").Buttons(b).OnAction = "OpenContainingFolderMac"
                    
                    Case Else
                
                        ' Do nothing
                
                End Select
            
            Case "Win"
            
                Select Case Sheets("Home").Buttons(b).Caption
                
                    Case "Open containing folder"
                    
                        Sheets("Home").Buttons(b).OnAction = "DoOpenContainingFolder"
                    
                    Case Else
                
                        ' Do nothing
                
                End Select
            
            Case Else ' Not Windows or Mac
            
                Select Case Sheets("Home").Buttons(b).Caption
                
                    Case "Open containing folder"
                    
                        Sheets("Home").Buttons(b).OnAction = "DummyMacro"
                    
                    Case Else
                
                        ' Do nothing
        
                End Select
        
        End Select
    
    Next
    
    ' --- Buttons: Home tab ENDS ---
    
    Debug.Print "--- Workbook_Open ENDS ---"

End Sub ' Workbook_Open
